```Go
// целочисленные типы
int8  // от -128 до 127 | 1 байт (8 бит)

int16 //  от -32768 до 32767 | 2 байта (16 бит)

int32 // от -2147483648 до 2147483647 | 4 байта (32 бита)

int64 // от –9 223 372 036 854 775 808 до 9 223 372 036 854 775 807 | 8 байт (64 бита)

uint8 // от 0 до 255 | 1 байт

byte // == uint8

uint16 // от 0 до 65535 | 2 байта

uint32 // от 0 до 4294967295 | 4 байта

uint64 // от 0 до 18 446 744 073 709 551 615 | 8 байт

byte // синоним типа uint8, от 0 до 255 | 1 байт

rune // синоним типа int32, от -2147483648 до 2147483647 | 4 байта

int // целое число со знаком, которое в зависимости от платформы
		// может занимать либо 4 байта, либо 8 байт.
    // То есть соответствовать либо int32, либо int64.

uint // целое беззнаковое число только без знака, которое,
		 // аналогично типу int, в зависимости о платформы
		 // может занимать либо 4 байта, либо 8 байт.
		 // То есть соответствовать либо uint32, либо uint64.
```

другие важные/интересные числовые типы

```Go
Для представления дробных чисел есть два типа:

float32 // число с плавающей точкой от 1.4*10-45 до 3.4*1038.
			  // Занимает в памяти 4 байта (32 бита)

float64 // число с плавающей точкой от 4.9*10-324 до 1.8*10308
        // занимает 8 байт.

// Существуют отдельные типы для представления комплексных чисел:
complex64 // вещественная и мнимая части представляют числа float32

complex128 // вещественная и мнимая части представляют числа float64
```

Строки
```go
// управляющие символы интерпретируются как управляющие по-умолчанию
var str string = "\t- Hello!\n"

// строка, в которой никакие символы не интрепретируются
strRaw := `The most ubiquitous flow-control symbols are \t and \r`

// байты (alias для uint8)
var char byte = '\x27' // 39

// руны для представления любых символов, которых нет в ascii (uint32)
var gigaAlphabet rune = 'кек лол тринитротолуол' // каждая руна занимает 4 байта

// (неизменяемы, нельзя по индексу что-то поменять)
```
